y_simulated = train_imputed['Target'].copy()

# Balance data using SMOTE (no internet so we balance manually)
df_balanced = pd.concat([
    train_imputed[y_simulated == 0].sample(300, random_state=42),
    train_imputed[y_simulated == 1]
])
X_balanced = df_balanced.drop(columns='Target')
# Reconstruct target from the dataframe instead of manually defining 600 values
X_balanced = df_balanced.drop(columns='Target')
y_balanced = df_balanced['Target']


# Train-test split
X_train_nn, X_val_nn, y_train_nn, y_val_nn = train_test_split(
    X_balanced, y_balanced, test_size=0.2, random_state=42
)

# Normalize inputs
from sklearn.preprocessing import StandardScaler
scaler = StandardScaler()
X_train_scaled = scaler.fit_transform(X_train_nn)
X_val_scaled = scaler.transform(X_val_nn)

# Build Neural Network with SGD optimizer
model = Sequential([
    Dense(64, activation='relu', input_shape=(X_train_scaled.shape[1],)),
    Dropout(0.3),
    Dense(32, activation='relu'),
    Dense(1, activation='sigmoid')
])

model.compile(optimizer=SGD(learning_rate=0.01), loss='binary_crossentropy', metrics=['accuracy'])

# Train the model
history = model.fit(X_train_scaled, y_train_nn, epochs=20, batch_size=32, verbose=0, validation_data=(X_val_scaled, y_val_nn))

# Predictions and Evaluation
y_pred_nn = model.predict(X_val_scaled).flatten()

# Function to train and evaluate different models
def build_and_evaluate_model(X_train, y_train, X_val, y_val, layers_config, optimizer, use_dropout=False, class_weights=None):
    model = Sequential()
    input_shape = X_train.shape[1]

    # Input layer
    model.add(Dense(layers_config[0], activation='relu', input_shape=(input_shape,)))
    if use_dropout:
        model.add(Dropout(0.3))

    # Hidden layers
    for units in layers_config[1:]:
        model.add(Dense(units, activation='relu'))
        if use_dropout:
            model.add(Dropout(0.3))

    # Output layer
    model.add(Dense(1, activation='sigmoid'))

    # Compile
    model.compile(optimizer=optimizer, loss='binary_crossentropy', metrics=['accuracy'])

    # Train
    model.fit(X_train, y_train, epochs=20, batch_size=32, verbose=0,
              validation_data=(X_val, y_val), class_weight=class_weights)

    # Predict
    y_pred_prob = model.predict(X_val).flatten()
    y_pred = (y_pred_prob > 0.5).astype(int)

    # Evaluate
    acc = accuracy_score(y_val, y_pred)
    cm = confusion_matrix(y_val, y_pred)
    report = classification_report(y_val, y_pred, output_dict=True)

    return {
        "accuracy": acc,
        "f1_score": report['1']['f1-score'],
        "precision": report['1']['precision'],
        "recall": report['1']['recall'],
        "confusion_matrix": cm,
        "report": report
    }
